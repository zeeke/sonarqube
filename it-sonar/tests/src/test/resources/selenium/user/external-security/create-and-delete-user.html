store javascript{new Date().getTime()} userId
click id=create-link-user
waitForVisible user_create_form
type user_login ${userId}
type user_name Name of ${userId}

type user_password_confirmation password
click commit

assertText info *User is created.*
click delete-${userId}
waitForVisible confirm-form
assertText css=.modal-body *Warning : are you sure to delete the user "Name of ${userId}"?*
clickAndWait confirm-submit
assertNotText users user-${userId}
waitForElementPresent Link=Log out
